    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const startBtn = document.getElementById('startBtn');
            const stopBtn = document.getElementById('stopBtn');
            const statusText = document.getElementById('statusText');
            const statusSpinner = document.getElementById('statusSpinner');
            const progressBar = document.getElementById('progressBar');
            const progressText = document.getElementById('progressText');
            const progressPercent = document.getElementById('progressPercent');
            const currentMovie = document.getElementById('currentMovie');
            const errorCount = document.getElementById('errorCount');
            const threadCount = document.getElementById('threadCount');
            const queueSize = document.getElementById('queueSize');
            const crawledCount = document.getElementById('crawledCount');
            const startTime = document.getElementById('startTime');
            const endTime = document.getElementById('endTime');
            const lastUpdate = document.getElementById('lastUpdate');
            const lastUpdatedTime = document.getElementById('lastUpdatedTime');
            
            let updateInterval;
            
            // 获取系统状态
            function fetchStatus() {
                fetch('/status')
                    .then(response => response.json())
                    .then(data => {
                        updateUI(data);
                    })
                    .catch(error => {
                        console.error('获取状态失败:', error);
                    });
            }
            
            // 更新UI
            function updateUI(data) {
                // 更新状态文本
                statusText.textContent = data.status_text;
                
                // 更新进度
                const processed = data.processed || 0;
                const total = data.total || 1;
                const percent = total > 0 ? Math.round((processed / total) * 100) : 0;
                
                progressBar.style.width = `${percent}%`;
                progressText.textContent = `${processed}/${total}`;
                progressPercent.textContent = `${percent}%`;
                
                // 更新状态指示器
                if (data.is_running) {
                    statusSpinner.classList.remove('d-none');
                    startBtn.disabled = true;
                    stopBtn.disabled = false;
                    progressBar.classList.add('progress-bar-animated');
                } else {
                    statusSpinner.classList.add('d-none');
                    startBtn.disabled = false;
                    stopBtn.disabled = true;
                    progressBar.classList.remove('progress-bar-animated');
                }
                
                // 更新统计数据
                errorCount.textContent = data.error_count || 0;
                threadCount.textContent = data.active_threads || 0;
                queueSize.textContent = data.queue_size || 0;
                crawledCount.textContent = data.crawled_count || 0;
                currentMovie.textContent = data.current_movie || '无';
                
                // 更新时间信息
                startTime.textContent = data.start_time || '-';
                endTime.textContent = data.end_time || '-';
                lastUpdate.textContent = data.last_update || '-';
                lastUpdatedTime.textContent = new Date().toLocaleTimeString();
            }
            
            // 开始爬取
            startBtn.addEventListener('click', function() {
                const movieLimit = document.getElementById('movieLimit').value;
                
                fetch('/start', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({ movie_limit: movieLimit })
                })
                .then(response => response.json())
                .then(data => {
                    if (data.status === 'started') {
                        showAlert('爬虫已启动！', 'success');
                    } else {
                        showAlert(data.message, 'warning');
                    }
                })
                .catch(error => {
                    console.error('启动失败:', error);
                    showAlert('启动爬虫失败，请检查服务器', 'danger');
                });
            });
            
            // 停止爬取
            stopBtn.addEventListener('click', function() {
                fetch('/stop', {
                    method: 'POST'
                })
                .then(response => response.json())
                .then(data => {
                    if (data.status === 'stopping') {
                        showAlert('正在停止爬虫...', 'info');
                    } else {
                        showAlert(data.message, 'warning');
                    }
                })
                .catch(error => {
                    console.error('停止失败:', error);
                    showAlert('停止爬虫失败', 'danger');
                });
            });
            
            // 显示提示信息
            function showAlert(message, type) {
                // 移除现有的提示
                const existingAlert = document.querySelector('.alert-dismissible');
                if (existingAlert) {
                    existingAlert.remove();
                }
                
                const alertDiv = document.createElement('div');
                alertDiv.className = `alert alert-${type} alert-dismissible fade show position-fixed top-0 start-50 translate-middle-x mt-3`;
                alertDiv.style.zIndex = '1050';
                alertDiv.innerHTML = `
                    ${message}
                    <button type="button" class="btn-close" data-bs-dismiss="alert" aria-label="Close"></button>
                `;
                
                document.body.appendChild(alertDiv);
                
                // 5秒后自动消失
                setTimeout(() => {
                    alertDiv.classList.remove('show');
                    setTimeout(() => alertDiv.remove(), 150);
                }, 5000);
            }
            
            // 初始化
            fetchStatus();
            updateInterval = setInterval(fetchStatus, 2000);
            
            // 清理定时器
            window.addEventListener('beforeunload', function() {
                clearInterval(updateInterval);
            });
        });
    </script>
